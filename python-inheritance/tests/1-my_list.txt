MyList = __import__('1-my_list').MyList

# Create an instance of MyList and add elements
my_list = MyList()
my_list.append(1)
my_list.append(4)
my_list.append(2)
my_list.append(3)
my_list.append(5)

# Check the content of the list
print(my_list)
# Expected output:
# [1, 4, 2, 3, 5]

# Call print_sorted method
my_list.print_sorted()
# Expected output (sorted list):
# [1, 2, 3, 4, 5]

# Check the original list again to ensure it is not modified
print(my_list)
# Expected output:
# [1, 4, 2, 3, 5]
